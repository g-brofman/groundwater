ama <- read_csv("ama.csv") %>% 
  clean_names()
## Parsed with column specification:
## cols(
##   YEAR = col_double(),

##   `PARENT WATER TYPE OR SECTOR` = col_character(),

##   QUANTITY = col_double()

ama_sorted <- ama %>% 
  group_by(ama, year) %>% 
  summarize(annual_use = sum(quantity))

#time series for individual AMAs
ggplot(ama_sorted,
       aes(x = year,
           y = annual_use,
           color = ama)) +
  geom_line()

#time series for all AMAs combined
ama_combined <- ama_sorted %>% 
  group_by(year) %>% 
  summarize(annual_use = sum(annual_use))

ggplot(ama_combined,
       aes(x = year,
           y = annual_use)) +
  geom_line()
